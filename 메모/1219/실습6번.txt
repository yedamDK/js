  <script>
    window.addEventListener("load", function() {
      //확인버튼이벤트
      btn.addEventListener("click", function() {
        //선택과목 수와 선택과목명을 div에 출력(다중선택이 가능함)
        let option = document.querySelectorAll("#subject option:checked");
        
        // console.log(option.length);
        // 하나라도 option(Array)에 값이 없으면 걸리겠죠?
        if(option.length === 0) {
          alert("선택 한 과목이 없습니다.");
          return;
        }

        // 하나라도 option(Array)에 값이 있으면 if가 안걸리고 여기로 고스란히 오심
        let text_array = [];
        for(let i = 0; i < option.length; i++) {
          text_array.push(option[i].value);
        }
        
        document.getElementById("result").innerHTML = `
          선택과목 수 : ${text_array.length}
          <br>
          선택과목 : ${text_array.join(", ")}
        `;
      });
    });
  </script>
